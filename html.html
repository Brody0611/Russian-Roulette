let milesTraveled = 0;
let food = 500; // Starting with some food
let health = 100; // 100 is good health
let gameActive = false;
let gameSpeed = "Normal";
const asciiDisplay = document.getElementById('ascii-display');
const statusMessage = document.getElementById('status-message');
const milesSpan = document.getElementById('miles');
const foodSpan = document.getElementById('food');
const healthSpan = document.getElementById('health');
const controlsDiv = document.getElementById('controls');
function updateDisplay(asciiArt, message) {
asciiDisplay.textContent = asciiArt;
statusMessage.textContent = message;
milesSpan.textContent = milesTraveled;
foodSpan.textContent = food;
healthSpan.textContent = health > 50 ? 'Good' : health > 20 ? 'Fair' : 'Poor';
}
function startGame() {
gameActive = true;
updateDisplay(wagonArt, "You are on the trail. Choose your action.");
controlsDiv.innerHTML =   ;
}
function endGame() {
gameActive = false;
updateDisplay(gameOverArt, "Game Over. Thanks for playing!");
controlsDiv.innerHTML = ;
}
function travel() {
if (!gameActive) return;
const distance = Math.floor(Math.random() * 50) + 20;
const foodUsed = 20;
milesTraveled += distance;
food -= foodUsed;
if (food <= 0) {
gameOver("You ran out of food and starved to death.");
return;
}
// Random events
if (Math.random() < 0.3) {
const event = randomEvent();
updateDisplay(wagonArt, You traveled ${distance} miles. ${event});
} else {
updateDisplay(wagonArt, You traveled ${distance} miles.);
}
if (milesTraveled >= 2000) { // Simple win condition
winGame();
}
}
function hunt() {
if (!gameActive) return;
const foodFound = Math.floor(Math.random() * 100) + 50;
food += foodFound;
health += 10; // hunting improves health slightly
if (health > 100) health = 100;
updateDisplay(huntArt, You went hunting and found ${foodFound} lbs of food.);
}
function rest() {
if (!gameActive) return;
const foodUsed = 10;
food -= foodUsed;
health += 20;
if (health > 100) health = 100;
if (food <= 0) {
gameOver("You ran out of food while resting and starved.");
return;
}
updateDisplay(restArt, "You rested and recovered your strength.");
}
function randomEvent() {
const events = [
"A sudden storm damaged your supplies!", food -= 50,
"You found an abandoned wagon with some food.", food += 30,
"One of your party members got sick (dysentery).", health -= 30,
"A friendly traveler shared some food and advice.", food += 20,
"Your wagon wheel broke.", milesTraveled -= 10,
];
const eventIndex = Math.floor(Math.random() * (events.length / 2)) * 2;
// Execute the associated action
eval(events[eventIndex + 1]);
return events[eventIndex];
}
function gameOver(message) {
gameActive = false;
updateDisplay(gameOverArt, Game Over: ${message});
controlsDiv.innerHTML = ;
}
function winGame() {
gameActive = false;
updateDisplay(winArt, "CONGRATULATIONS! You have reached Oregon!");
controlsDiv.innerHTML = ;
}
function learnAboutTrail() {
updateDisplay(infoArt, "The Oregon Trail was a major migration route... Try to survive!");
}
